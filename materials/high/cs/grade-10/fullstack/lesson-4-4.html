שיעור 4.4 — הוספה, רישום והתחברות

INSERT, Registration, Login עם SqlCommandBuilder

1. הוספת רשומה (INSERT) בשיטה הלא מקושרת

// יצירת שורה חדשה ב-DataTable DataRow newRow = dt.NewRow(); newRow["Username"] = Username; newRow["Password"] = Password; newRow["Email"] = Email; dt.Rows.Add(newRow); // SqlCommandBuilder מייצר INSERT אוטומטי SqlCommandBuilder builder = new SqlCommandBuilder(adapter); adapter.Update(dt);
SqlCommandBuilder

מייצר אוטומטית פקודות INSERT, UPDATE ו-DELETE על סמך פקודת ה-SELECT המקורית. חוסך כתיבת SQL ידנית!

2. רישום (Registration)

לפני הוספת משתמש חדש — חייבים לבדוק שאין כבר משתמש עם אותו שם:

public void OnPostRegister() { // בדיקת כפילות DataRow[] found = dt.Select($"Username = '{Username}'"); if (found.Length > 0) { Message = "שם המשתמש תפוס! בחרו שם אחר."; return; } // הוספת משתמש חדש DataRow newRow = dt.NewRow(); newRow["Username"] = Username; newRow["Password"] = Password; newRow["FullName"] = FullName; dt.Rows.Add(newRow); SqlCommandBuilder builder = new SqlCommandBuilder(adapter); adapter.Update(dt); Message = "נרשמת בהצלחה!"; }
