3. מבוא ל-RegExp

ביטוי רגולרי (Regular Expression) הוא דפוס לחיפוש טקסט. כמו "חפש מספרי טלפון" או "בדוק אם זה אימייל תקין".

// יצירת RegExp let pattern = /hello/; // literal let pattern2 = new RegExp("hello"); // constructor // test() — מחזיר true/false console.log(/hello/.test("say hello world")); // true console.log(/hello/.test("say goodbye")); // false // עוגנים: ^ (תחילה), $ (סוף) console.log(/^hello/.test("hello world")); // true — מתחיל ב-hello console.log(/world$/.test("hello world")); // true — מסתיים ב-world

5. דפוסים מעשיים

// טלפון ישראלי: 05X-XXXXXXX let phonePattern = /^0[2-9]\d{7,8}$/; console.log(phonePattern.test("0521234567")); // true console.log(phonePattern.test("12345")); // false // אימייל (פשוט) let emailPattern = /^[\w.-]+@[\w.-]+\.\w{2,}$/; console.log(emailPattern.test("user@mail.com")); // true console.log(emailPattern.test("not-email")); // false // סיסמה חזקה: 8+ תווים, אות גדולה, קטנה, ספרה function isStrongPassword(pass) { return pass.length >= 8 && /[A-Z]/.test(pass) && /[a-z]/.test(pass) && /\d/.test(pass); }
